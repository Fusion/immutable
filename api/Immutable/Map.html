def each #

Returns an iterator over the map entries, returning a Tuple of the key and value. The order of iteration is not specified.

map = Immutable::Map.new({"foo" => "bar", "baz" => "qux"})
iterator = map.each

entry = iterator.next
entry[0] # => "foo"
entry[1] # => "bar"

entry = iterator.next
entry[0] # => "baz"
entry[1] # => "qux"
